### Creating a Blog
```typescript
// Define blog post type
const blogPostType = {
  name: 'Blog Post',
  apiId: 'blog-post',
  fields: [
    {
      name: 'title',
      type: 'shortText',
      required: true
    },
    {
      name: 'content',
      type: 'richText'
    }
  ]
};

// Create post
const post = await api.createContent('blog-post', {
  title: 'My First Post',
  content: '# Hello World'
});
```

## Integration

### Frontend Integration
```typescript
// React example
function BlogPost({ id }) {
  const [post, setPost] = useState(null);
  
  useEffect(() => {
    api.getContent('blog-post', id)
      .then(setPost);
  }, [id]);
  
  return post ? (
    <article>
      <h1>{post.title}</h1>
      <div>{post.content}</div>
    </article>
  ) : null;
}
```

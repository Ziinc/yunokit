# Field Types Reference

## Text Fields

### Short Text
```typescript
{
  type: 'shortText',
  maxLength?: number,
  pattern?: string
}
```

### Long Text
```typescript
{
  type: 'longText',
  markdown?: boolean,
  maxLength?: number
}
```

### Rich Text
```typescript
{
  type: 'richText',
  allowedFeatures?: string[],
  maxLength?: number
}
```

## Numeric Fields

### Integer
```typescript
{
  type: 'integer',
  min?: number,
  max?: number
}
```

### Decimal
```typescript
{
  type: 'decimal',
  precision?: number,
  scale?: number
}
```

## Date & Time Fields

### Date
```typescript
{
  type: 'date',
  format?: 'YYYY-MM-DD'
}
```

### DateTime
```typescript
{
  type: 'datetime',
  withTimezone?: boolean
}
```

## Media Fields

### Image
```typescript
{
  type: 'image',
  allowedTypes?: string[],
  maxSize?: number
}
```

### File
```typescript
{
  type: 'file',
  allowedTypes?: string[],
  maxSize?: number
}
```

## Structured Fields

### JSON
```typescript
{
  type: 'json',
  schema?: JSONSchema
}
```

### Array
```typescript
{
  type: 'array',
  items: FieldType,
  minItems?: number,
  maxItems?: number
}
```

## Reference Fields

### Single Reference
```typescript
{
  type: 'reference',
  contentType: string,
  allowNull?: boolean
}
```

### Multiple Reference
```typescript
{
  type: 'multiReference',
  contentType: string,
  minItems?: number,
  maxItems?: number
}
```

## Custom Fields

### Component
```typescript
{
  type: 'component',
  component: string,
  props?: Record<string, any>
}
```

### Custom
```typescript
{
  type: 'custom',
  validator?: (value: any) => boolean,
  renderer?: string
}
```
